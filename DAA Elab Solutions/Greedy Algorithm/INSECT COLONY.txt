#include<stdio.h>
#include<string.h>
 

    int main()
{
  int t;
    scanf("%d",&t);
    int i;
    for(i=0;i<t;i++)
    {
        int n;
        scanf("%d",&n);
        long long int j,count=0,ele;
        for(j=0;j<n;j++)
        {
            scanf("%lld",&ele);
            count = count^ele;
        }
        if(count%2==0) printf("Yes\n");
        else printf("No\n");
    }
 
 return 0;
    }
