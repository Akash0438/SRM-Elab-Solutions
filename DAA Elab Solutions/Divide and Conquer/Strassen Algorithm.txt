#include <iostream>
using namespace std;
int main()
{
int avi[50][50],b[50][50],c,d,i,size,k,p[50][50],sum=0;
  
  cin>>size;
  
  for(c=0;c<size;c++ ) 
  { 
    for(d=0;d<size;d++) 
    { 
      cin>>avi[c][d];
    } 
  } 
   for(c=0;c<size;c++ ) 
  { 
    for(d=0;d<size;d++) 
    { 
      cin>>b[c][d];
    } 
  } 
   for(c=0;c<size;c++ ) 
  { 
    for(d=0;d<size;d++) 
    { 
      for(k=0;k<size;k++) 
      { 
       sum=sum+avi[c][k]*b[k][d];
        }
      p[c][d]=sum;

    } 
  } 
     for(c=0;c<size;c++ )

    for(d=0;d<size;d++)

    } cout<<endl;
  } 
 return 0;
}
