#include <iostream> 
using namespace std; 
  
int maxSum(int arr[], int n, int k) 
{ 
    if (n < k) 
    { 
       cout << "Invalid"; 
       return -1; 
    } 
  
    int res = 0; 
    for (int i=0; i<k; i++) 
       res += arr[i]; 
  

    int curr_sum = res; 
    for (int i=k; i<n; i++) 
    { 
       curr_sum += arr[i] - arr[i-k]; 
       res = max(res, curr_sum); 
    } 
  
    return res; 
} 

int main() 
{ 
    
    int k = 2; 
    int n,i,j,a ;
  cin>>a;
  for(i=0;i<a;i++) { 
    cin>>n;
    int arr[n];
    for(j=0;j<n;j++) { 
      cin>>arr[j]; } 
    cout << maxSum(arr, n, k)<<endl; }
    return 0; 
}
