#include <iostream>
using namespace std;
class A
{
public:
int x;
void getxval()
{
cin>>x;
}
};
class B
{
public:
int y;
void getyval()
{
cin>>y;
}
};
class C:public A,public B
{
public:
void sum()
{
cout<<"Sum = "<<x+y<<endl;
}
void mul()
{
cout<<"Product="<<x*y;
}
};
int main()
{
C obj;
obj.getxval();
obj.getyval();
obj.sum();
obj.mul();
return 0;
}
