#include <iostream>
using namespace std;
int main()
{
int n;
cin>>n;
int a[n];
for(int i=0; i<n; i++)
cin>>a[i];
for(int i=0; i<n; i++)
for(int j=i+1; j<n; j++)
if(a[i]>a[j])
{
int t=a[i];
a[i]=a[j];
a[j]=t;
}
for(int i=0; i<n; i++)
if(a[i]%2)
cout<<a[i]<<endl;
for(int i=0; i<n; i++)
if(!(a[i]%2))
cout<<a[i]<<endl;
return 0;
}
